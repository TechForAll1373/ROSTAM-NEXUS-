```bash
# Install a base system first (assumes archinstall or manual)
# Then choose your DE:

# For Qt (KDE Plasma)
sudo pacman -S plasma-meta kde-applications-meta sddm
sudo systemctl enable sddm

# For GTK (GNOME)
sudo pacman -S gnome gnome-extra gdm
sudo systemctl enable gdm

# Reboot
reboot

# OpenVPN & NetworkManager integration
sudo pacman -S networkmanager networkmanager-openvpn openvpn
sudo systemctl enable --now NetworkManager

# WireGuard
sudo pacman -S wireguard-tools

# Sing‑box (from AUR)
yay -S sing-box

# Tor
sudo pacman -S tor
sudo systemctl enable --now tor.service

# Clash Meta (from AUR)
yay -S clash-meta

# Set proxy environment variables (temporary)
export http_proxy="http://127.0.0.1:7890"
export https_proxy="http://127.0.0.1:7890"

# Tor Browser (from AUR)
yay -S tor-browser

# Firefox (hardened)
sudo pacman -S firefox

# Brave
yay -S brave-bin

# Ungoogled Chromium
yay -S ungoogled-chromium

# Yay (AUR helper) – if not installed
sudo pacman -S --needed base-devel git
git clone https://aur.archlinux.org/yay.git
cd yay
makepkg -si

# Flatpak (for more apps)
sudo pacman -S flatpak xdg-utils
flatpak remote-add --if-not-exists flathub https://flathub.org/repo/flathub.flatpakrepo

# Wine (Windows compatibility)
sudo pacman -S wine winetricks
yay -S wine-gecko wine-mono

# Waydroid (Android in a container)
sudo pacman -S waydroid
sudo waydroid init
systemctl enable --now waydroid-container.service

sudo pacman -S networkmanager networkmanager-openvpn openvpn
sudo systemctl enable --now NetworkManager

sudo pacman -S wireguard-tools

# Install yay first if not present
sudo pacman -S --needed base-devel git
git clone https://aur.archlinux.org/yay.git
cd yay
makepkg -si

# Then install sing-box
yay -S sing-box

yay -S clash-meta

export http_proxy="http://127.0.0.1:7890"
export https_proxy="http://127.0.0.1:7890"
export all_proxy="socks5://127.0.0.1:1080"
